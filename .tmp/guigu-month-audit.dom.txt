  <script>
    function sleep(ms) {
      return new Promise(function (resolve) { setTimeout(resolve, ms); });
    }

    async function waitFor(selector, timeout) {
      var start = Date.now();
      while (Date.now() - start < timeout) {
        var node = document.querySelector(selector);
        if (node) return node;
        await sleep(50);
      }
      return null;
    }

    async function createCharacter() {
      await waitFor('[data-action="new"]', 4000);
      document.querySelector('[data-action="new"][data-slot="0"]')?.click();
      await sleep(120);

      document.getElementById('create-name').value = '月策审计';
      document.getElementById('create-next')?.click();
      await sleep(120);

      document.querySelector('.spirit-root-option')?.click();
      await sleep(80);
      document.getElementById('create-next')?.click();
      await sleep(120);

      document.querySelector('.sect-option')?.click();
      await sleep(80);
      document.getElementById('create-next')?.click();
      await sleep(120);

      var firstTalent = document.querySelector('.talent-option:not(.disabled)');
      firstTalent?.click();
      await sleep(100);
      var secondTalent = document.querySelector('.talent-option:not(.disabled):not(.selected)');
      secondTalent?.click();
      await sleep(60);
      document.getElementById('create-next')?.click();
      await sleep(120);

      document.querySelector('.personality-option')?.click();
      await sleep(80);
      document.getElementById('create-next')?.click();
      await sleep(120);

      document.getElementById('create-confirm')?.click();
      await sleep(400);
    }

    async function runAudit() {
      await createCharacter();
      var chooser = await waitFor('[data-month-focus]', 3000);
      var chooserVisibleBefore = !!chooser;
      var focusButtons = Array.prototype.slice.call(document.querySelectorAll('[data-month-focus]'));
      var adventureBtn = document.querySelector('[data-month-focus="adventure"]') || focusButtons[0];
      adventureBtn?.click();
      await sleep(240);

      var game = window.__guiguGame;
      var bannerFocus = document.querySelector('[data-month-plan-focus]')?.getAttribute('data-month-plan-focus') || null;
      var bannerDays = Number(document.querySelector('[data-month-plan-days]')?.getAttribute('data-month-plan-days') || 0);
      var result = {
        gameExposed: !!game,
        chooserVisibleBefore: chooserVisibleBefore,
        focusCount: focusButtons.length,
        selectedFocus: game && game.state && game.state.monthPlan ? game.state.monthPlan.focus : null,
        bannerFocus: bannerFocus,
        bannerDays: bannerDays,
        errors: window.__errors
      };
      result.pass =
        result.gameExposed &&
        result.chooserVisibleBefore &&
        result.focusCount >= 4 &&
        result.selectedFocus === 'adventure' &&
        result.bannerFocus === 'adventure' &&
        result.bannerDays === 30 &&
        result.errors.length === 0;
      document.getElementById('result').textContent = JSON.stringify(result, null, 2);
    }

    window.addEventListener('load', function () {
      setTimeout(function () {
        runAudit().catch(function (error) {
          window.__errors.push(String(error));
          document.getElementById('result').textContent = JSON.stringify({ pass: false, errors: window.__errors }, null, 2);
        });
      }, 150);
    });
  </script>
