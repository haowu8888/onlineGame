  <script>
    function sleep(ms) {
      return new Promise(function (resolve) { setTimeout(resolve, ms); });
    }

    async function runAudit() {
      document.getElementById('ct-btn-start')?.click();
      await sleep(180);
      document.querySelector('.ct-class-card')?.click();
      await sleep(320);

      const firstChoice = document.querySelector('.ct-tower-node.available[data-node-type="event"]') ||
        document.querySelector('.ct-tower-node.available[data-node-type="rest"]') ||
        document.querySelector('.ct-tower-node.available');
      firstChoice?.click();
      await sleep(260);

      const eventChoice = document.querySelector('#ct-event-choices [data-choice-id="leave"]') ||
        document.querySelector('#ct-event-choices .ct-event-choice:last-child') ||
        document.querySelector('#ct-event-choices .ct-event-choice');
      if (eventChoice) {
        eventChoice.click();
        await sleep(260);
      }

      const restChoice = document.querySelector('#ct-rest-choices .ct-rest-choice');
      if (restChoice) {
        restChoice.click();
        await sleep(260);
      }

      const eliteNode = document.querySelector('.ct-tower-node.available[data-node-type="elite"]');
      eliteNode?.click();
      await sleep(350);

      const affixEl = document.querySelector('.ct-enemy-affix, .ct-status-badge.affix, [data-enemy-affix]');
      const game = window.__cardtowerGame;
      if (game && game.battle && game.battle.enemies) {
        game.battle.enemies.forEach(function (enemy) { enemy.hp = 0; });
        game.battle.battleWon();
      }
      await sleep(900);

      const relicChoices = document.querySelectorAll('#ct-relic-choices .ct-relic-choice');
      relicChoices[0]?.click();
      await sleep(320);

      const nextNodes = document.querySelectorAll('.ct-tower-node.available');
      const result = {
        eliteReached: !!eliteNode,
        affixVisible: !!affixEl,
        relicChoiceCount: relicChoices.length,
        nextAvailableCount: nextNodes.length,
        gameExposed: !!game,
        errors: window.__errors
      };
      result.pass = result.eliteReached && result.affixVisible && result.relicChoiceCount >= 3 && result.nextAvailableCount >= 1 && result.gameExposed && result.errors.length === 0;
      document.getElementById('result').textContent = JSON.stringify(result, null, 2);
    }

    window.addEventListener('load', function () {
      setTimeout(function () {
        runAudit().catch(function (error) {
          window.__errors.push(String(error));
          document.getElementById('result').textContent = JSON.stringify({ pass: false, errors: window.__errors }, null, 2);
        });
      }, 120);
    });
