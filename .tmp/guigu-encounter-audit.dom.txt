    function sleep(ms) {
      return new Promise(function (resolve) { setTimeout(resolve, ms); });
    }

    async function waitFor(selector, timeout) {
      var start = Date.now();
      while (Date.now() - start < timeout) {
        var node = document.querySelector(selector);
        if (node) return node;
        await sleep(50);
      }
      return null;
    }

    async function createCharacter() {
      await waitFor('[data-action="new"]', 4000);
      document.querySelector('[data-action="new"][data-slot="0"]')?.click();
      await sleep(120);
      document.getElementById('create-name').value = '机缘审计';
      document.getElementById('create-next')?.click();
      await sleep(120);
      document.querySelector('.spirit-root-option')?.click();
      await sleep(80);
      document.getElementById('create-next')?.click();
      await sleep(120);
      document.querySelector('.sect-option')?.click();
      await sleep(80);
      document.getElementById('create-next')?.click();
      await sleep(120);
      var firstTalent = document.querySelector('.talent-option:not(.disabled)');
      firstTalent?.click();
      await sleep(100);
      var secondTalent = document.querySelector('.talent-option:not(.disabled):not(.selected)');
      secondTalent?.click();
      await sleep(60);
      document.getElementById('create-next')?.click();
      await sleep(120);
      document.querySelector('.personality-option')?.click();
      await sleep(80);
      document.getElementById('create-next')?.click();
      await sleep(120);
      document.getElementById('create-confirm')?.click();
      await sleep(360);
      document.querySelector('[data-month-focus="adventure"]')?.click();
      await sleep(240);
    }

    async function runAudit() {
      await createCharacter();
      document.querySelector('.guigu-tab[data-tab="adventure"]')?.click();
      await sleep(260);

      var game = window.__guiguGame;
      var cards = Array.prototype.slice.call(document.querySelectorAll('[data-encounter-card-id]'));
      var firstCard = cards[0] || null;
      var firstChoices = firstCard ? firstCard.querySelectorAll('[data-encounter-choice]') : [];
      firstChoices[0]?.click();
      await sleep(260);

      var result = {
        gameExposed: !!game,
        cardCount: cards.length,
        firstChoiceCount: firstChoices.length,
        deckCount: game && game.state && game.state.encounterDeck ? game.state.encounterDeck.length : 0,
        encountersResolved: game && game.state && game.state.monthlyStats ? game.state.monthlyStats.encountersResolved : 0,
        lastEncounterId: game && game.state && game.state.lastEncounterResult ? game.state.lastEncounterResult.encounterId : null,
        rerenderedCardCount: document.querySelectorAll('[data-encounter-card-id]').length,
        errors: window.__errors
      };
      result.pass =
        result.gameExposed &&
        result.cardCount >= 3 &&
        result.firstChoiceCount >= 2 &&
        result.deckCount >= 3 &&
        result.encountersResolved >= 1 &&
        !!result.lastEncounterId &&
        result.rerenderedCardCount >= 3 &&
        result.errors.length === 0;
      document.getElementById('result').textContent = JSON.stringify(result, null, 2);
    }

    window.addEventListener('load', function () {
      setTimeout(function () {
        runAudit().catch(function (error) {
          window.__errors.push(String(error));
          document.getElementById('result').textContent = JSON.stringify({ pass: false, errors: window.__errors }, null, 2);
        });
      }, 150);
    });
  </script>
