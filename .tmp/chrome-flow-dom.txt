  <script>
    function wait(ms) {
      return new Promise(function (resolve) {
        setTimeout(resolve, ms);
      });
    }

    function getPhase2State() {
      return JSON.parse(localStorage.getItem('cultivation_save_1') || '{}').phase2Adventure || {};
    }

    async function ensureRouteChoices() {
      if (document.querySelectorAll('.phase2-route-card').length) return;
      const startBtn = document.getElementById('btn-phase2-adventure-start');
      if (startBtn) {
        startBtn.click();
        await wait(120);
      }
    }

    async function rerollRoutes(routeBias) {
      const biasBtn = document.querySelector('[data-route-bias="' + routeBias + '"]');
      if (biasBtn) {
        biasBtn.click();
        await wait(120);
      }
      await ensureRouteChoices();
    }

    async function pickRouteByType(types, routeBiases, maxAttempts) {
      for (let attempt = 0; attempt < maxAttempts; attempt += 1) {
        await ensureRouteChoices();
        const state = getPhase2State();
        const choice = (state.currentChoices || []).find(function (item) {
          return types.includes(item.type);
        });
        if (choice) {
          document.querySelector('[data-route-choice="' + choice.choiceId + '"]')?.click();
          await wait(180);
          return choice;
        }
        await rerollRoutes(routeBiases[attempt % routeBiases.length]);
      }
      return null;
    }

    async function finishBattle() {
      for (let attempt = 0; attempt < 8; attempt += 1) {
        const backBtn = document.querySelector('[data-action="back"]');
        if (backBtn) {
          backBtn.click();
          await wait(180);
          return true;
        }
        const attackBtn = document.querySelector('[data-action="attack"]');
        if (!attackBtn) break;
        attackBtn.click();
        await wait(180);
      }
      return false;
    }

    window.addEventListener('load', function () {
      (async function () {
        await wait(80);
        document.querySelector('[data-enter="1"]')?.click();
        await wait(120);
        document.querySelector('.cult-tab[data-tab="battle"]')?.click();
        await wait(120);

        const result = {
          immediateChoices: [],
          blessingBeforeBattle: null,
          blessingAfterBattle: null,
          battleChoiceType: null,
          battleResolved: false,
          finalChoiceCount: 0,
          finalStep: 0,
          finalBlessingCount: 0,
          errors: window.__errors
        };

        const firstImmediate = await pickRouteByType(['event', 'rest', 'harvest'], ['steady', 'greedy'], 4);
        if (firstImmediate) {
          result.immediateChoices.push(firstImmediate.type);
        }

        const secondImmediate = await pickRouteByType(['event', 'rest', 'harvest'], ['steady', 'greedy'], 4);
        if (secondImmediate) {
          result.immediateChoices.push(secondImmediate.type);
        }

        result.blessingBeforeBattle = document.querySelectorAll('[data-route-blessing]').length;
        if (result.blessingBeforeBattle) {
          document.querySelector('[data-route-blessing]')?.click();
          await wait(120);
        }

        const battleChoice = await pickRouteByType(['battle'], ['gamble', 'steady', 'greedy'], 8);
        result.battleChoiceType = battleChoice ? battleChoice.type : null;
        if (battleChoice) {
          result.battleResolved = await finishBattle();
        }

        result.blessingAfterBattle = document.querySelectorAll('[data-route-blessing]').length;
        const finalState = getPhase2State();
        result.finalChoiceCount = document.querySelectorAll('.phase2-route-card').length;
        result.finalStep = finalState.step || 0;
        result.finalBlessingCount = (finalState.blessings || []).length;
        result.pass =
          result.immediateChoices.length === 2 &&
          result.blessingBeforeBattle >= 1 &&
          result.battleChoiceType === 'battle' &&
          result.battleResolved &&
          result.blessingAfterBattle >= 1 &&
          result.finalChoiceCount === 3 &&
          result.finalStep >= 3 &&
          result.finalBlessingCount >= 1 &&
          result.errors.length === 0;

        document.getElementById('result').textContent = JSON.stringify(result, null, 2);
      })();
    });
