    function sleep(ms) {
      return new Promise(function (resolve) { setTimeout(resolve, ms); });
    }

    async function waitFor(selector, timeout) {
      var start = Date.now();
      while (Date.now() - start < timeout) {
        var node = document.querySelector(selector);
        if (node) return node;
        await sleep(50);
      }
      return null;
    }

    async function createCharacter() {
      await waitFor('[data-action="new"]', 4000);
      document.querySelector('[data-action="new"][data-slot="0"]')?.click();
      await sleep(120);
      document.getElementById('create-name').value = '宗门审计';

    async function runAudit() {
      await createCharacter();
      document.querySelector('.guigu-tab[data-tab="sect"]')?.click();
      await sleep(200);

      var game = window.__guiguGame;
      var oldOrder = game && game.state && game.state.sectPressure ? game.state.sectPressure.currentOrder : null;
      if (game && oldOrder) {
        game.state.gold = Math.max(game.state.gold, 9999);
        game.state.day = 30;
        game.state.monthlyStats = game.state.monthlyStats || {};
        game.state.monthlyStats[oldOrder.metric] = oldOrder.target;
        game.advanceDays(1);
      }
      await sleep(260);
      document.querySelector('.guigu-tab[data-tab="sect"]')?.click();
      await sleep(200);

      var nextOrder = game && game.state && game.state.sectPressure ? game.state.sectPressure.currentOrder : null;
      var summary = game && game.state ? game.state.lastMonthSummary : null;
      var result = {
        gameExposed: !!game,
        orderId: oldOrder ? oldOrder.id : null,
        pressureValue: game && game.state && game.state.sectPressure ? game.state.sectPressure.pressure : null,
        summarySuccess: !!(summary && summary.sectOrderSuccess === true),
        nextOrderId: nextOrder ? nextOrder.id : null,
        pressureVisible: !!document.querySelector('[data-sect-pressure-value]'),
        orderVisible: !!document.querySelector('[data-sect-order-id]'),
        errors: window.__errors
      };
      result.pass =
        result.gameExposed &&
        !!result.orderId &&
        result.pressureVisible &&
        result.orderVisible &&
        result.summarySuccess &&
        !!result.nextOrderId &&
        result.nextOrderId !== result.orderId &&
        result.errors.length === 0;
      document.getElementById('result').textContent = JSON.stringify(result, null, 2);
    }

    window.addEventListener('load', function () {
      setTimeout(function () {
        runAudit().catch(function (error) {
          window.__errors.push(String(error));
          document.getElementById('result').textContent = JSON.stringify({ pass: false, errors: window.__errors }, null, 2);
        });
      }, 150);
    });
