      document.getElementById('ct-btn-start')?.click();
      await sleep(180);
      document.querySelector('.ct-class-card')?.click();
      await sleep(320);

      const firstChoice = document.querySelector('.ct-tower-node.available[data-node-type="event"]') ||

        document.querySelector('.ct-tower-node.available');

      const eventChoice = document.querySelector('#ct-event-choices [data-choice-id="leave"]') ||
        document.querySelector('#ct-event-choices .ct-event-choice:last-child') ||
        document.querySelector('#ct-event-choices .ct-event-choice');
      if (eventChoice) {
        eventChoice.click();
        await sleep(260);

      const restChoice = document.querySelector('#ct-rest-choices .ct-rest-choice');
      if (restChoice) {
        restChoice.click();
        await sleep(260);

      eliteNode?.click();

      const game = window.__cardtowerGame;
      if (game && game.battle && game.battle.enemies) {
        game.battle.enemies.forEach(function (enemy) { enemy.hp = 0; });
        game.battle.battleWon();

      const relicChoices = document.querySelectorAll('#ct-relic-choices .ct-relic-choice');
